% list of symbols entries
% To use the symbols in equations and to get a proper reference to the list of symbols \gls{sym:a} needs to be used for instance so as an example (makes no sence at all):
%
% \begin{equation} (or align instead of equation)
% 	\gls{sym:a} = \gls{sym:e} + \gls{sym:gamma}
% \end{equation}

\newglossaryentry{sym:a}
{
type=symbol, % entry should be in the list of symbols!
name={\ensuremath{a}}, % Put the symbol here in dollar signs
description={Semi-major axis}, % A brief description of this entry (to appear in the glossary).
symbol={$\left[m\right]$}, % put the unit here.... kind of weird, i know 
sort=a, % for correct sorting type the full name of the symbol here
parent=romanletter % for sorting purposes, use romanletter or greekletter
}
